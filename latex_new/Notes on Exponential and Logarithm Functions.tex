  \emph{IMPORTANT IDEAS:}
  \par
    $\ln x$ is the inverse of $e^x$, that is, $ e^{\ln x}=x.$
  \end{minipage}
}

\vspace{10 mm}

\fbox{%

  \begin{minipage}{\textwidth}
  \emph{Properties of Logarithm Functions}
  \begin{enumerate}
  \item $\ln(a\cdot b)=\ln a+\ln b$.
  \item $\ln(\frac{a}{b})=\ln a-\ln b$.
  \item $\ln (a^b)=b\ln a.$
  \end{enumerate}
  \end{minipage}
}
\vspace{10 mm}

\fbox{%

  \begin{minipage}{\textwidth}
  \emph{Properties of Exponential Functions}
  \begin{enumerate}
  \item $e^{a+b}=e^a\cdot e^b$.
  \item $e^{a-b}=\frac{a}{b}$.
  \item $(e^a)^b=e^{ab}$
  \end{enumerate}
  \end{minipage}
}
